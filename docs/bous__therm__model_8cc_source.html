164 double BousThermModel::f_Kint (double aqbot, double zedge, double *Tin) {
165 
166  //if the bottom of the aquifer is above the water table, no mobility
167  if (aqbot >= zedge) return(0.0);
168 
169  //find the deepest partially thawed cell
170  long idxl = point_inside(ze, aqbot, Nz+1);
171  //find the highest partially saturated cell
172  long idxt = point_inside(ze, zedge, Nz+1);
173 
174  //integrate hydraulic conductivity
175  double Kint = 0.0;
176  //freezing point and water table are inside a single cell
177  if (idxl == idxt) {
178  Kint = (zedge - aqbot)*f_K(perm[idxl], Tin[idxl]);
179  //freezing point and water table are in different cells
180  } else if (idxt > idxl) {
181  //lowest cell containing the freezing point
182  Kint += (ze[idxl+1] - aqbot)*f_K(perm[idxl], Tin[idxl]);
183  //intermediate cells
184  for (long i=idxl+1; i<idxt; i++)
185  Kint += delz[i]*f_K(perm[i], Tin[i]);
186  //cell containing the water table
187  Kint += (zedge - ze[idxt])*f_K(perm[idxt], Tin[idxt]);
188  }
189 
190  return(Kint);
191 }

206 double BousThermModel::f_dHdt (double fl, double fr, double po, double delx) {
207  /*
208  Compute the time derivative of the water table for a cell, given fluxes
209  and other quantities
210  inputs:
211  fl - flux into the cell from the left edge (m^2/s)
212  fr - flux out of the cell from the right edge (m^2/s)
213  re - recharge into the cell (m/s)
214  po - porosity at the current water table level (dimensionless)
215  dx - width of the cell (m)
216  returns:
217  dHdt - the time derivative of the water table level (m/s)
218  */
219 
220  return( (fl - fr)/(delx*po) );

248 void BousThermModel::ode_fun (double *solin, double *fout) {
249 
250  //----------------------------------------------------------
251 
252  //aliases for the inputs
253  Hin = solin;
254  for (long j=0; j<Nx+1; j++)
255  Tin[j] = solin + Nx + Nz*j;
256  //aliases for the outputs
257  dHdt = fout;
258  for (long j=0; j<Nx+1; j++)
259  dTdt[j] = fout + Nx + Nz*j;
260 
261  //----------------------------------------------------------
262 
263  //update surface temperature
264  for (long j=0; j<Nx+1; j++)
265  Tsurf[j] = f_surf_temp(get_t(), htope[j], stg->Ts0, stg->Tsf, stg->Tsgam, stg->TsLR);
266 
267  //hydraulic gradients and edge values
268  //left edge
269  gradH[0] = 0.0;
270  Hedge[0] = Hin[0];
271  //interior edges
272  for (long j=1; j<Nx; j++) {
273  gradH[j] = (Hin[j] - Hin[j-1])/(xc[j] - xc[j-1]);
274  Hedge[j] = Hin[j-1] + gradH[j]*(xe[j] - xc[j-1]);
275  //check the edge value is below the surface
276  if ( Hedge[j] > ztope[j] )
277  Hedge[j] = ztope[j];
278  }
279  //right edge
280  gradH[Nx] = 0.0;
281  Hedge[Nx] = Hin[Nx-1];
282 
283  //big parallel loop computes
284  // thermal gradients
285  // aquifer bottom points
286  // saturation fractions of water and ice
287  // updated thermal capacities
288  // thermal fluxes
289  // vertically integrated hydraulic conductivities
290  // hydraulic fluxes
291  #pragma omp parallel for
292  for (long j=0; j<Nx+1; j++) {
293  //geothermal gradient
294  gradT[j][0] = -stg->fTgeo/ktherm;
295  //interior edges
296  for (long i=1; i<Nz; i++)
297  gradT[j][i] = (Tin[j][i] - Tin[j][i-1])/(zc[i] - zc[i-1]);
298  //surface gradient
299  gradT[j][Nz] = (Tsurf[j] - Tin[j][Nz-1])/(delz[Nz-1]/2.0);
300  //find the aquifer bottom
301  aqbot[j] = f_aquifer_bottom(Tin[j], Tsurf[j]);
302  //update saturation fractions
303  update_sat(Hedge[j] - ztope[j], aqbot[j], Tin[j], wsat[j], isat[j]);
304  //compute thermal fluxes
305  for (long i=0; i<Nz+1; i++)
306  qT[j][i] = f_qT(gradT[j][i]);
307  //update thermal capacities and compute thermal time derivatives
308  for (long i=0; i<Nz; i++) {
309  captherm[j][i] = f_captherm(poro[i], Tin[j][i], wsat[j][i], isat[j][i]);
310  dTdt[j][i] = f_dTdt(qT[j][i], qT[j][i+1], delz[i], captherm[j][i]);
311  }
312  //compute hydraulic conductivity for the edge
313  Kint[j] = f_Kint(aqbot[j], Hedge[j] - ztope[j], Tin[j]);
314  //compute GW flux for the edge
315  qH[j] = f_qH(gradH[j], Kint[j]);
316  }
317 
318  //----------------------------------------------------------
319 
320  //compute hydraulic time derivatives
321  for (long j=0; j<Nx; j++)
322  dHdt[j] = f_dHdt(qH[j], qH[j+1], poro[point_inside(ze, Hin[j] - ztopc[j], Nz+1)], delx[j]);
323 
324 }

326 void BousThermModel::update_evaporation () {
327  //take water out of the top as necessary, as evaporation
328  for (long j=0; j<Nx; j++) {
329  if ( H[j] > ztopc[j] ) {
330  //evaporation rate by area (m^2/s)
331  evap[j] = (H[j] - ztopc[j])*delx[j]*poro_surf/get_dt();
332  //evaporation rate by depth (m/s)
333  evapw[j] = (H[j] - ztopc[j])*poro_surf/get_dt();
334  //contribution to cumulative evaporation array (m^2)
335  cumevap[j] += evap[j]*get_dt();
336  //remove excess water
337  H[j] = ztopc[j];
338  } else {
339  evap[j] = 0.0;
340  evapw[j] = 0.0;
341  }
342  }
343 }
