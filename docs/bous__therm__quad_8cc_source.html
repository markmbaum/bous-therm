88  *y = 0.0;
89  for (int i=0; i<n; i++) {

100 void QuadGaussLegendre::init_gq (double *x, double *w, int n) {
101  /*
102  Compute the abscissas and weights for Gauss-Legendre quadrature, following
103  the algorithm explained in chapter 4 of:
104  Press, William H., et al. Numerical recipes 3rd edition: The art of scientific computing. Cambridge university press, 2007.
105  */
106 
107  //abscissas are symmetric about zero, only compute half directly
108  int m = (n + 1)/2;
109  //Newton tolerance
110  double tol = 1e-14;
111  //Newton cutoff
112  int maxiter = 100;
113  //things for Newton iteration
114  double xx, xprev, y, dy;
115  int niter;
116  //pi!
117  double pi = 3.1415926535897932384626433832795;
118 
119  //compute the abscissas and weights
120  for (int i=0; i<m; i++) {
121  //initial estimate
122  xx = cos(pi*(i + 0.75)/(n + 0.5));
123  //refine with Newton's method
124  niter = 0;
125  xprev = 0.0;
126  P(xx, n, &y, &dy);
127  while ( (fabs(y) > tol) && (fabs(xx - xprev) > tol) ) {
128  //update x
129  xprev = xx;
130  xx -= y/dy;
131  //update y
132  P(xx, n, &y, &dy);
133  //update iteration count
134  niter++;
135  if (niter > maxiter) {
136  printf("QUADRATURE ERROR: Newton's method failed to converge in initialization of Gauss-Legendre quadrature class");
137  exit(EXIT_FAILURE);
138  }
139  }
140  //store the abscissa
141  x[i] = -xx;
142  x[n-1-i] = xx;
143  //compute the weight
144  w[i] = 2.0/((1.0 - xx*xx)*dy*dy);
145  w[n-1-i] = w[i];
146  }
147 }
